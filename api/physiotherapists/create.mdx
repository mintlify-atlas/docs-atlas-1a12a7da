## Request Body

<ParamField body="nombre" type="string" required>
  Physiotherapist's first name
</ParamField>

<ParamField body="apellidoPaterno" type="string" required>
  Physiotherapist's paternal last name
</ParamField>

<ParamField body="apellidoMaterno" type="string" required>
  Physiotherapist's maternal last name
</ParamField>

<ParamField body="dni" type="string" required>
  National identification number (must be unique)
</ParamField>

<ParamField body="correo" type="string" required>
  Email address (must be unique)
</ParamField>

<ParamField body="password" type="string" required>
  Account password
</ParamField>

<ParamField body="cumpleanios" type="string">
  Date of birth
</ParamField>

<ParamField body="genero" type="string">
  Gender
</ParamField>

<ParamField body="img" type="string">
  Profile image URL
</ParamField>

<ParamField body="celular" type="string">
  Mobile phone number
</ParamField>

<ParamField body="direccion" type="string">
  Physical address
</ParamField>

<ParamField body="especialidad" type="string">
  Medical specialty (e.g., sports therapy, pediatric therapy)
</ParamField>

<ParamField body="colegiatura" type="string">
  Professional license number
</ParamField>

<ParamField body="universidad" type="string">
  University where the degree was obtained
</ParamField>

<ParamField body="descProfesional" type="string">
  Professional description and biography
</ParamField>

<ParamField body="estado" type="string" default="activo">
  Account status. Possible values: `activo`, `inactivo`
</ParamField>
