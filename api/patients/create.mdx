---
title: "Create Patient"
api: "POST /api/paciente"
description: "Create a new patient with medical history and allergy information"
---

## Request Body

<ParamField body="nombre" type="string" required>
  Patient's first name
</ParamField>

<ParamField body="apellidoPaterno" type="string" required>
  Patient's paternal last name
</ParamField>

<ParamField body="apellidoMaterno" type="string">
  Patient's maternal last name
</ParamField>

<ParamField body="dni" type="string" required>
  Patient's national identification number (DNI). Must be unique.
</ParamField>

<ParamField body="cumpleanios" type="string">
  Patient's date of birth
</ParamField>

<ParamField body="genero" type="string">
  Patient's gender
</ParamField>

<ParamField body="img" type="string">
  URL or path to patient's profile image
</ParamField>

<ParamField body="celular" type="string">
  Patient's phone number
</ParamField>

<ParamField body="direccion" type="string">
  Patient's address
</ParamField>

<ParamField body="correo" type="string" required>
  Patient's email address. Must be unique.
</ParamField>

<ParamField body="password" type="string" required>
  Patient's password for authentication
</ParamField>

<ParamField body="estado" type="string" default="activo">
  Patient's account status. Possible values: `activo`, `inactivo`
</ParamField>

<ParamField body="historialMedico" type="string[]">
  Array of medical history entries for the patient
</ParamField>

<ParamField body="alergias" type="string[]">
  Array of known allergies for the patient
</ParamField>
