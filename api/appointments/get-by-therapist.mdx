---
title: 'Get Appointments by Therapist'
api: 'GET /api/cita/{id}'
description: 'Retrieve all appointments for a specific therapist'
---

This endpoint returns all appointments associated with a specific therapist ID.

## Path Parameters

<ParamField path="id" type="string" required>
  Therapist ID (MongoDB ObjectId)
</ParamField>

## Response

Returns an array of appointment objects for the specified therapist.

<ResponseField name="appointments" type="array">
  Array of appointment objects
  
  <ResponseField name="_id" type="string">
    Unique appointment identifier
  </ResponseField>
  
  <ResponseField name="paciente" type="object">
    Patient information (populated)
    <ResponseField name="_id" type="string">
      Patient ID
    </ResponseField>
    <ResponseField name="nombre" type="string">
      Patient first name
    </ResponseField>
    <ResponseField name="apellidoPaterno" type="string">
      Patient paternal surname
    </ResponseField>
    <ResponseField name="apellidoMaterno" type="string">
      Patient maternal surname
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="terapeuta" type="object">
    Therapist information (populated)
    <ResponseField name="_id" type="string">
      Therapist ID
    </ResponseField>
    <ResponseField name="nombre" type="string">
      Therapist first name
    </ResponseField>
    <ResponseField name="apellidoPaterno" type="string">
      Therapist paternal surname
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="fecha" type="string">
    Appointment date
  </ResponseField>
  
  <ResponseField name="hora" type="string">
    Appointment time
  </ResponseField>
  
  <ResponseField name="motivo" type="string">
    Reason for the appointment
  </ResponseField>
  
  <ResponseField name="precio" type="number">
    Appointment price
  </ResponseField>
  
  <ResponseField name="estado" type="string">
    Appointment status (pendiente, confirmada, cancelada, completada)
  </ResponseField>
</ResponseField>
