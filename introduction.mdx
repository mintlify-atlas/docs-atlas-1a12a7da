<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick Start</h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">Get your API up and running in minutes</p>
  
  <Steps>
    <Step title="Clone and Install">
      Clone the repository and install dependencies:
      
      ```bash
      git clone https://github.com/LuisCondori98/back-fisioterapia.git
      cd back-fisioterapia
      npm install
      ```
    </Step>
    
    <Step title="Configure Environment">
      Create a `.env` file with your MongoDB connection and JWT secret:
      
      ```bash
      MONGODB_URI=mongodb://localhost:27017/fisioterapia
      JWT_SECRET=your_secret_key
      FACEBOOK_ID=your_facebook_app_id
      FACEBOOK_SECRET=your_facebook_secret
      ```
      
      <Note>See the [Setup Environment](/guides/setup-environment) guide for complete configuration options.</Note>
    </Step>
    
    <Step title="Start the Server">
      Run the development server:
      
      ```bash
      npm run dev
      ```
      
      The API will be available at `http://localhost:3000`
    </Step>
    
    <Step title="Create Your First User">
      Make your first API call to create a user:
      
      ```bash
      curl -X POST http://localhost:3000/api/user \
        -H "Content-Type: application/json" \
        -d '{
          "nombre": "Juan",
          "apellidoPaterno": "Pérez",
          "apellidoMaterno": "García",
          "dni": "12345678",
          "correo": "juan@example.com",
          "password": "securepassword123",
          "rol": "paciente"
        }'
      ```
      
      <Accordion title="Example Response">
        ```json
        {
          "status": "success",
          "data": {
            "_id": "507f1f77bcf86cd799439011",
            "nombre": "Juan",
            "apellidoPaterno": "Pérez",
            "correo": "juan@example.com",
            "rol": "paciente",
            "estado": "activo"
          }
        }
        ```
      </Accordion>
    </Step>
  </Steps>
</div>
